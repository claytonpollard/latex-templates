% Cloze
%%% `\ShowClozetrue' to enable, `\ShowClozefalse' to disable
\newif\ifShowCloze
\ShowClozefalse
\newcommand{\Cloze}[1]{
    \ifShowCloze
        \dotuline{\parbox[b]{\widthof{\LARGE\textbf{#1}}}{ \vphantom{\vspace{1ex}} \centering %
        {\color{OrangeRed}  {\protect #1}} %
        }}
    \else
        \dotuline{\parbox[b]{\widthof{\LARGE\textbf{#1}}}{ \vphantom{\vspace{1ex}}\centering %
            \hfill
            }
        }
    \fi
}
